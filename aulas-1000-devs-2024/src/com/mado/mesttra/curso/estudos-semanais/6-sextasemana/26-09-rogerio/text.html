        } else if (dia == 31 && (mes == 2 || mes == 4 || mes == 6 || mes == 9 || mes == 11)) {
            System.out.print("Um dado é invalido pois não existe 31 no mes + " + mes);

        } else if (dia == 30 && mes == 2) {

            // e executado quando qtdeMorange é > 5

            // e executado quando qtdeMorange não é > 5
